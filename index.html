1. Línea con Interpolación RGB y Grosor Dinámico
Algoritmo DDA/Bresenham con transición de color y control de grosor
                    
                        // Algoritmo DDA con interpolación de color
                        function drawLineWithColor(x1, y1, x2, y2, color1, color2, thickness) {
                            const points = [];
                            const dx = x2 - x1;
                            const dy = y2 - y1;
                            const steps = Math.max(Math.abs(dx), Math.abs(dy));
                            
                            // Interpolación de color
                            const colors = interpolateRGB(color1, color2, steps);
                            
                            const xInc = dx / steps;
                            const yInc = dy / steps;
                            
                            let x = x1, y = y1;
                            
                            for (let i = 0; i <= steps; i++) {
                                points.push({
                                    x: Math.round(x),
                                    y: Math.round(y),
                                    color: colors[i]
                                });
                                x += xInc;
                                y += yInc;
                            }
                            
                            return applyThickness(points, thickness);
                        }

                        function interpolateRGB(color1, color2, steps) {
                            const colors = [];
                            const [r1, g1, b1] = color1;
                            const [r2, g2, b2] = color2;
                            
                            for (let i = 0; i <= steps; i++) {
                                const t = i / steps;
                                const r = Math.round(r1 + (r2 - r1) * t);
                                const g = Math.round(g1 + (g2 - g1) * t);
                                const b = Math.round(b1 + (b2 - b1) * t);
                                colors.push(`rgb(${r}, ${g}, ${b})`);
                            }
                            
                            return colors;

                        }

📖 Explicación del Algoritmo

Objetivo: Dibujar una línea entre dos puntos con transición suave de color y grosor configurable.


🔧 Características implementadas:

  • Algoritmo DDA: Calcula incrementos fraccionarios para suavizar la línea
  • Interpolación RGB: Transición lineal entre dos colores cualquiera
  • Grosor dinámico: Expansión de píxeles alrededor de la línea base
  • Control interactivo: Modificación en tiempo real de parámetros

2. Algoritmo de Elipse con Punto Medio
Implementación eficiente usando simetría de cuadrantes
// Algoritmo de elipse por punto medio
function drawEllipseMidpoint(xc, yc, a, b) {
    const points = [];
    let x = 0;
    let y = b;
    
    // Región 1
    let d1 = (b * b) - (a * a * b) + (0.25 * a * a);
    
    while ((a * a * (y - 0.5)) > (b * b * (x + 1))) {
        plotEllipsePoints(xc, yc, x, y, points);
        
        if (d1 < 0) {
            d1 += b * b * (2 * x + 3);
        } else {
            d1 += b * b * (2 * x + 3) + a * a * (-2 * y + 2);
            y--;
        }
        x++;
    }
    
    // Región 2
    let d2 = (b * b * (x + 0.5) * (x + 0.5)) + 
             (a * a * (y - 1) * (y - 1)) - 
             (a * a * b * b);
    
    while (y >= 0) {
        plotEllipsePoints(xc, yc, x, y, points);
        
        if (d2 < 0) {
            d2 += b * b * (2 * x + 2) + a * a * (-2 * y + 3);
            x++;
        } else {
            d2 += a * a * (-2 * y + 3);
        }
        y--;
    }
    
    return points;
}

3. Figura Fractal Tipo Flor
Patrón recursivo usando circunferencias y elipses concéntricas
// Algoritmo fractal floral
function createFlowerFractal(x, y, baseRadius, depth, maxDepth) {
    if (depth > maxDepth) return;
    
    // Dibujar círculo base
    drawCircle(x, y, baseRadius);
    
    // Dibujar pétalos (elipses rotadas)
    const petals = 6;
    for (let i = 0; i < petals; i++) {
        const angle = (i * 2 * Math.PI) / petals;
        const petalX = x + Math.cos(angle) * baseRadius * 1.5;
        const petalY = y + Math.sin(angle) * baseRadius * 1.5;
        
        drawEllipse(petalX, petalY, baseRadius * 0.6, baseRadius * 0.3, angle);
        
        // Llamada recursiva para cada pétalo
        createFlowerFractal(petalX, petalY, baseRadius * 0.4, depth + 1, maxDepth);
    }
}
